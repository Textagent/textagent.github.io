  method: "POST",
  headers: { "Content-Type": "application/json", "x-api-key": API_KEY },

    model: "claude-sonnet-4-5-20250929",
    system: memoryText,
    messages: [{ role: "user", content: "..." }],
    max_tokens: 1024
  })
});
